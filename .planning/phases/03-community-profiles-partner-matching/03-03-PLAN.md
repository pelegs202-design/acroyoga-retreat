Query members using Drizzle conditional filters:
```typescript
import { and, eq, ilike, isNotNull, ne } from 'drizzle-orm';

const members = await db
  .select({
    id: user.id,
    name: user.name,
    image: user.image,
    city: user.city,
    role: user.role,
    level: user.level,
    bio: user.bio,
    skills: user.skills,
  })
  .from(user)
  .where(
    and(
      city ? ilike(user.city, `%${city}%`) : undefined,
      role ? eq(user.role, role) : undefined,
      level ? eq(user.level, level) : undefined,
      isNotNull(user.tosAcceptedAt),  // Only show users who completed signup
      ne(user.id, session.user.id),    // Don't show self in results
    )
  )
  .limit(50);
```

```typescript
  if (!session) redirect({ href: '/sign-in', locale });

  const [member] = await db
    .select()
    .from(user)
    .where(eq(user.id, userId))
    .limit(1);

  if (!member) notFound();

}
```
